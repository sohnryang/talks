% definitions in the body so the figure is \includestandalone-safe
\definecolor{memfill}{HTML}{BFD7F2}%
\definecolor{memdraw}{HTML}{2563EB}%
\definecolor{compfill}{HTML}{FFD9A8}%

\begin{tikzpicture}[x=0.62cm, font=\sffamily, lbl/.style={font=\sffamily\small}]
\def\bh{0.34}\def\Mw{1.5}\def\Cw{2.5} % Mem narrower, Comp wider -> compute-bound

% Compute bars run back-to-back (the bottleneck). Each next load starts exactly
% when the current compute starts: Mem_{k+1}.start = Comp_k.start (double-buffered).
% step / Mem-start / Comp-start / y-center
\foreach \k/\ls/\cs/\yc in {1/0/1.5/3.85, 2/1.5/4/2.95, 3/4/6.5/2.05, 4/6.5/9/1.15} {
    \node[lbl, anchor=east] at (-0.4,\yc) {Step \k};
    \draw[gray!25, line width=0.4pt] (0,\yc) -- (11.7,\yc);
    % Mem (blue)
    \fill[memfill] (\ls,\yc-\bh) rectangle (\ls+\Mw,\yc+\bh);
    \node[lbl, memdraw!60!black] at (\ls+\Mw/2,\yc) {Mem};
    % Comp (orange)
    \fill[compfill] (\cs,\yc-\bh) rectangle (\cs+\Cw,\yc+\bh);
    \node[lbl, font=\sffamily\footnotesize, compdraw!50!black] at (\cs+\Cw/2,\yc) {Comp};
}

% time axis + ticks at the Comp-start positions
\draw[-{Stealth[length=6pt]}, line width=0.8pt] (0,0.3) -- (12.0,0.3);
\foreach \t in {0,1.5,4,6.5,9} { \draw[line width=0.6pt] (\t,0.22) -- (\t,0.38); }
\node[lbl, anchor=west] at (12.1,0.3) {Time};

% dashed lines drawn last (on top of all bars). The first line marks the start of
% the first compute (x=1.5 -> Comp1); the rest sit at each Comp-end and connect up
% to the finishing corner of the PREVIOUS step's Comp bar.
\foreach \xf/\ycp in {1.5/3.85, 4/3.85, 6.5/2.95, 9/2.05} {
    \draw[densely dashed, gray!55, line width=0.5pt] (\xf,0.3) -- (\xf,\ycp-\bh);
}

\end{tikzpicture}
