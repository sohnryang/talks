% memory bar with on-bar label (#1 x0 #2 x1 #3 yc #4 opacity #5 label)
\newcommand{\mbar}[5]{%
  \fill[memhue, fill opacity=#4] (#1,#3-\bh) rectangle (#2,#3+\bh);
  \draw[memhue, draw opacity=0.5, line width=0.3pt] (#1,#3-\bh) rectangle (#2,#3+\bh);
  \pgfmathsetmacro{\mx}{(#1+#2)/2}%
  \ifdim#4 pt>0.62pt\relax
    \node[font=\sffamily\scriptsize, text=white] at (\mx,#3) {#5};%
  \else
    \node[font=\sffamily\scriptsize, text=memhue!80!black] at (\mx,#3) {#5};%
  \fi}
% compute bar with on-bar label (#1 x0 #2 x1 #3 yc #4 opacity #5 label)
\newcommand{\cbar}[5]{%
  \fill[comphue, fill opacity=#4] (#1,#3-\bh) rectangle (#2,#3+\bh);
  \draw[comphue, draw opacity=0.5, line width=0.3pt] (#1,#3-\bh) rectangle (#2,#3+\bh);

  \else
    \node[font=\sffamily\scriptsize, text=comphue!85!black] at (\mx,#3) {#5};%
  \fi}
